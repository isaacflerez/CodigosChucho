#include <Servo.h>

Servo servoX;
Servo servoY;

void setup() {
  servoX.attach(6);  // Conecta el servo X al pin 6
  servoY.attach(7);  // Conecta el servo Y al pin 7
}

void loop() {
  // Lee los valores analógicos del joystick
  int xValue = analogRead(A0);  // Lee el eje X (VRx) en el pin A0
  int yValue = analogRead(A1);  // Lee el eje Y (VRy) en el pin A1

  // Mapea los valores a los rangos del servo (ajusta según tus servos)
  int servoXPos = map(xValue, 0, 1023, 0, 180);
  int servoYPos = map(yValue, 0, 1023, 0, 180);

  // Mueve los servos a las posiciones calculadas
  servoX.write(servoXPos);
  servoY.write(servoYPos);

  delay(20); 
}
